    ```typescript
    export interface VerificationDateRange {
      targetMonth: string;  // YYYYMM
      startDay: string;     // DD (zero-padded)

    }

     * @returns 月別の日付範囲配列。レコードなしの場合は null。
     */

      records: TranscriptionRecord[]
    ): VerificationDateRange[] | null
    ```

    - Group by YYYYMM
    - For each month group: find min day and max day
    - Return array of `VerificationDateRange` sorted by targetMonth
    - Return null if no qualifying records
