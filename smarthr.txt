// 部署の階層表現から部署名の配列を取得
// dept: 部署の入れ子表現
// level: 出力要素数
function flattenNestedDeptObj(dept, level) {
  const flattenNestedDept = (dept, maxLevel) => {
    if (!dept) return [];                   // 親がない
    if (maxLevel == 1) return [dept.name];  // 階層が深すぎ
    return [...flattenNestedDept(dept.parent, maxLevel - 1), dept.name];
  }
  // 　levelに指定された個数を返却
  const deptArray = flattenNestedDept(dept, 3);
  while (deptArray.length < level) deptArray.push(""); // 長さを調整
  return deptArray;
};

/**

function getAuthHeader() {
  return {
    'Authorization': 'Bearer ' + ACCESS_TOKEN,
  };
}

// URLのパラメタを追加（存在するものは上書き)
function addOrReplaceUrlParameter(url, params) {
  const [base, qstr] = url.split('?');

  // params に含まれるキーを除外
  const keyVals = (qstr || "").split("&").filter(kv => !kv.split("=")[0] in params);
  const appendedKeyVals = Object.keys(params).map(key => encodeURI(key) + "=" + encodeURI(params[key]));

  return base + "?" + keyVals.concat(appendedKeyVals).join('&');
}

/**
 * 全一覧取得
 * ページ分割された全てのページの結果をまとめて返却
 * @param { string } url
 * @param { object }
 */
function getAllAPIReuslts(url, params = {}) {
  let page = 0;
  const perPage = 100;
  let results = [];
  let recCount  = 0;
  do {
    page++;
    const pageUrl = addOrReplaceUrlParameter(url, {
      ...params,
      page,
      per_page: perPage
    });

    const res = UrlFetchApp.fetch(pageUrl, {
      method: "get",
      headers: getAuthHeader(),
      muteHttpExceptions: false,
    });

    if (res.getResponseCode() != 200) {
      console.log(res);
      throw new Error("API アクセス失敗 code:" + res.getResponseCode() + res.getContentText());
    }
    results = results.concat(JSON.parse(res.getContentText()));
    recCount = res.getHeaders()["x-total-count"]
  } while(perPage * page < recCount)

  return results;
}
